    Test 1 — EXT-01: custom doneToolSchema
    ```typescript
    it('uses caller-provided doneToolSchema instead of _findingsSchema', async () => {
        const replySchema = z.object({ reply: z.string() });
        const rc = createMockRemoteCommands();

        // Mock LLM to emit a done-tool call matching the custom schema
        mockGenerateText.mockResolvedValue({
            finishReason: 'tool-calls',
            toolCalls: [{ toolName: 'submitResult', args: { reply: 'This is my answer.' } }],
            steps: [],
            text: '',
            usage: { promptTokens: 0, completionTokens: 0, totalTokens: 0 },
            response: { messages: [] },
        });

        const output = await runAgentLoop(
            {
                model: {} as any,
                systemPrompt: 'You are a helpful assistant.',
                userPrompt: 'What is 2+2?',
                doneToolSchema: replySchema,
                maxSteps: 2,
            },
            {
                remoteCommands: rc,
            },
        );

        // The done-tool schema was the custom one — the output should have the 'reply' field
        // (exact assertion depends on how AgentLoopOutput.findings is populated from custom schema)
        expect(mockGenerateText).toHaveBeenCalled();
        const callArgs = mockGenerateText.mock.calls[0][0];
        // Confirm the tools object contains submitResult built from replySchema (not _findingsSchema)
        expect(callArgs.tools).toBeDefined();
        // The loop was called with the custom schema — no review-specific fields required
    });
    ```

    Test 2 — EXT-02: initialMessages seeded at step 0
    ```typescript
    it('seeds initialMessages before user turn in message history', async () => {
        const rc = createMockRemoteCommands();
        const priorTurn: CoreMessage = {
            role: 'assistant',
            content: 'I analyzed the code and found no issues.',
        };

        mockGenerateText.mockResolvedValue({
            finishReason: 'stop',
            toolCalls: [],
            steps: [],
            text: '{}',
            usage: { promptTokens: 0, completionTokens: 0, totalTokens: 0 },
            response: { messages: [] },
        });

        await runAgentLoop(
            {
                model: {} as any,
                systemPrompt: 'System',
                userPrompt: 'Follow up question',
                initialMessages: [priorTurn],
                maxSteps: 1,
            },
            { remoteCommands: rc },
        );

        expect(mockGenerateText).toHaveBeenCalled();
        const callArgs = mockGenerateText.mock.calls[0][0];
        // When initialMessages provided, generateText is called with messages array
        expect(callArgs.messages).toBeDefined();
        const roles = callArgs.messages.map((m: any) => m.role);
        // system -> assistant (prior turn) -> user
        expect(roles).toEqual(['system', 'assistant', 'user']);
        expect(callArgs.messages[1].content).toBe('I analyzed the code and found no issues.');
    });
    ```
