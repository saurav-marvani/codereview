    ```typescript
    import { Injectable } from '@nestjs/common';
    import { createLogger } from '@kodus/flow';
    import type { ModelMessage } from 'ai';
    import { ConversationThreadRepository } from '../repositories/conversation-thread.repository';
    import type {
        ConversationTurn,
        IConversationSessionManager,
    } from '../../domain/contracts/conversation-session-manager.contract';

    /** Maximum turns returned by materializeInitialMessages. Older turns are dropped (not summarized). */
    const MAX_HISTORY_TURNS = 20;

    @Injectable()
    export class ConversationSessionManager implements IConversationSessionManager {
        private readonly logger = createLogger(ConversationSessionManager.name);

        constructor(
            private readonly repository: ConversationThreadRepository,
        ) {}

        async load(prKey: string): Promise<ConversationTurn[]> {
            const doc = await this.repository.findByPrKey(prKey);
            return doc?.turns ?? [];
        }

        async appendTurn(prKey: string, turn: ConversationTurn): Promise<void> {
            await this.repository.appendTurn(prKey, turn);
        }

        async materializeInitialMessages(
            prKey: string,
        ): Promise<ModelMessage[]> {
            const turns = await this.load(prKey);

            if (turns.length > MAX_HISTORY_TURNS) {
                this.logger.warn({
                    message: 'Conversation history truncated',
                    context: ConversationSessionManager.name,
                    serviceName: ConversationSessionManager.name,
                    metadata: {
                        prKey,
                        totalTurns: turns.length,
                        kept: MAX_HISTORY_TURNS,
                    },
                });
            }

            const recentTurns = turns.slice(-MAX_HISTORY_TURNS);

            return recentTurns.map(
                (t): ModelMessage => ({
                    role: t.role,
                    content: t.content,
                }),
            );
        }
    }
    ```

    Note: `ModelMessage` from `ai` accepts `role: 'user' | 'assistant'` which matches `ConversationTurn.role` exactly. Do not cast or coerce.
