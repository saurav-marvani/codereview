    ```typescript
    import { z } from 'zod';
    import { createLogger } from '@kodus/flow';
    import {
        runAgentLoop,
        type AgentLoopSecrets,
    } from '@libs/code-review/infrastructure/agents/llm/agent-loop';
    import type { ModelMessage } from 'ai';
    import type { LanguageModel } from 'ai';
    import type { SandboxInstance } from '@libs/sandbox/domain/contracts/sandbox.provider';
    import type { IConversationSessionManager, ConversationTurn } from '../../domain/contracts/conversation-session-manager.contract';

    const logger = createLogger('runConversationLoop');

    export interface ConversationLoopInput {
        model: LanguageModel;
        systemPrompt: string;
        userPrompt: string;
        prKey: string;
        /**
         * Acquired sandbox from SandboxLeaseManager. runConversationLoop does NOT
         * acquire or release this lease — that is the caller's responsibility.
         * Caller must wrap in try/finally: finally { await leaseManager.release(leaseId) }
         */
        sandbox: SandboxInstance;
        sessionManager: IConversationSessionManager;
        /**
         * Memory tools from buildConversationMemoryTools — spread into additionalTools.
         * Pass {} when memory tools are not available (self-hosted without KodyRulesModule).
         */
        memoryTools?: Record<string, any>;
        maxSteps?: number;
        agentName?: string;
        /** Full name for repo context, e.g. "org/repo". Used by buildAgentTools for cross-file ops. */
        repositoryFullName?: string;
        telemetryMetadata?: import('@libs/code-review/infrastructure/agents/llm/agent-loop').LangfuseTelemetryMetadata;
        byokConfig?: AgentLoopSecrets['byokConfig'];
        byokProvider?: string;
    }

    export interface ConversationLoopOutput {
        /** Plain text reply from the agent — never CodeSuggestion[]. */
        reply: string;
        steps: number;
        toolCalls: Array<{
            tool: string;
            args: Record<string, unknown>;
            result?: string;
        }>;
    }

    /**
     * Done-tool schema for conversation: expects { reply: string }.
     * This overrides the default _findingsSchema ({ suggestions: [] }) via EXT-01.
     */
    const CONVERSATION_DONE_SCHEMA = z.object({ reply: z.string() });

    /**
     * Run a single @kody conversation turn through the review agent runtime.
     *
     * Sandbox ownership: this function receives an already-acquired sandbox.
     * The caller is responsible for acquire() before calling and release() in a
     * finally block after calling.
     *
     * NullSandbox path: when sandbox.type === 'null', passes remoteCommands:undefined
     * to runAgentLoop, which sets isSelfContained=true and runs a single LLM step.
     * Memory tools (additionalTools) still work in this path.
     */
    export async function runConversationLoop(
        input: ConversationLoopInput,
        extraSecrets?: Partial<AgentLoopSecrets>,
    ): Promise<ConversationLoopOutput> {
        // NullSandbox detection: type===null means no E2B sandbox available.
        // Passing remoteCommands:undefined causes buildAgentTools to return {}
        // and the loop runs single-shot (isSelfContained=true).
        // NEVER pass NULL_SANDBOX_INSTANCE.remoteCommands — it has silent stubs.
        const remoteCommands =
            input.sandbox.type === 'null'
                ? undefined
                : (input.sandbox as any).remoteCommands;

        const initialMessages = await input.sessionManager.materializeInitialMessages(
            input.prKey,
        );

        logger.log({
            message: 'Starting conversation loop',
            context: 'runConversationLoop',
            serviceName: 'runConversationLoop',
            metadata: {
                prKey: input.prKey,
                sandboxType: input.sandbox.type,
                historyTurns: initialMessages.length,
                hasMemoryTools: Object.keys(input.memoryTools ?? {}).length > 0,
            },
        });

        const output = await runAgentLoop(
            {
                model: input.model,
                systemPrompt: input.systemPrompt,
                userPrompt: input.userPrompt,
                doneToolSchema: CONVERSATION_DONE_SCHEMA,
                initialMessages,
                additionalTools: input.memoryTools ?? {},
                maxSteps: input.maxSteps ?? 15,
                agentName: input.agentName ?? 'kodus-conversation-agent',
                repositoryFullName: input.repositoryFullName,
                telemetryMetadata: input.telemetryMetadata,
                // Conversation skips review-specific passes (coverage-recovery, verify)
                skipHeavyPasses: true,
                skipSynthesisRescue: true,
            },
            {
                remoteCommands,
                byokConfig: input.byokConfig,
                byokProvider: input.byokProvider,
                ...extraSecrets,
            },
        );

        // Extract reply from the done-tool schema output.
        // output.findings has shape { reply: string } when CONVERSATION_DONE_SCHEMA is used.
        // Fall back to output.text when running in self-contained mode (NullSandbox path).
        const reply: string =
            (output.findings as any)?.reply ?? output.text ?? '';

        // Persist the assistant reply to the thread.
        const assistantTurn: ConversationTurn = {
            role: 'assistant',
            content: reply,
            timestamp: new Date(),
            toolCalls: output.toolCalls.map((tc) => ({
                tool: tc.tool ?? tc.toolName ?? 'unknown',
                args: tc.args,
                result: tc.result,
            })),
        };
        await input.sessionManager.appendTurn(input.prKey, assistantTurn);

        logger.log({
            message: 'Conversation loop complete',
            context: 'runConversationLoop',
            serviceName: 'runConversationLoop',
            metadata: {
                prKey: input.prKey,
                steps: output.steps,
                toolCallCount: output.toolCalls.length,
                replyLength: reply.length,
            },
        });

        return {
            reply,
            steps: output.steps,
            toolCalls: output.toolCalls.map((tc) => ({
                tool: tc.tool ?? tc.toolName ?? 'unknown',
                args: tc.args,
                result: tc.result,
            })),
        };
    }
    ```
