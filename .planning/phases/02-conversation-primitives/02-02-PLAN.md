    **File 1: `libs/conversation/infrastructure/repositories/conversation-thread.repository.ts`**

    Mirror the atomic `$push` pattern from `libs/platformData/infrastructure/adapters/repositories/pullRequests.repository.ts:923`.

    ```typescript
    import { Injectable } from '@nestjs/common';
    import { InjectModel } from '@nestjs/mongoose';
    import type { Model } from 'mongoose';
    import { ConversationThreadModel } from './schemas/conversation-thread.model';
    import type { ConversationTurn } from '../../domain/contracts/conversation-session-manager.contract';

    @Injectable()
    export class ConversationThreadRepository {
        constructor(
            @InjectModel(ConversationThreadModel.name)
            private readonly threadModel: Model<ConversationThreadModel>,
        ) {}

        async findByPrKey(
            prKey: string,
        ): Promise<ConversationThreadModel | null> {
            return this.threadModel.findById(prKey).lean();
        }

        /**
         * Atomically append one turn to the thread for prKey.
         * $push is document-level atomic — two concurrent appends both succeed.
         * $setOnInsert sets createdAt only when inserting a new doc (upsert).
         */
        async appendTurn(prKey: string, turn: ConversationTurn): Promise<void> {
            const now = new Date();
            await this.threadModel.findOneAndUpdate(
                { _id: prKey },
                {
                    $push: { turns: turn },
                    $set: { updatedAt: now },
                    $setOnInsert: { createdAt: now },
                },
                { upsert: true },
            );
        }
    }
    ```
