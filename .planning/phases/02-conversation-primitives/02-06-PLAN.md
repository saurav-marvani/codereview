    ```typescript
    // test/unit/conversation/memory-regression.spec.ts

    jest.mock('@kodus/flow', () => ({
        createLogger: () => ({
            log: jest.fn(),
            warn: jest.fn(),
            error: jest.fn(),
            debug: jest.fn(),
        }),
    }));

    import { buildConversationMemoryTools } from '@libs/conversation/infrastructure/services/conversation-tools.factory';

    describe('Memory creation regression (TEST-03)', () => {
        const ORG_ID = 'org-regression-test';
        const TEAM_ID = 'team-regression-test';

        const mockKodyRulesService = {
            createOrUpdateMemory: jest.fn(),
            findMemories: jest.fn(),
        };

        beforeEach(() => {
            jest.clearAllMocks();
        });

        describe('KODUS_CREATE_MEMORY', () => {
            it('explicit @kody remember — creates memory and returns action:created with link', async () => {
                mockKodyRulesService.createOrUpdateMemory.mockResolvedValue({
                    action: 'created',
                    requiresApproval: false,
                    link: 'https://app.kodus.io/rules/org-regression-test/memories',
                    rule: {
                        uuid: 'rule-uuid-123',
                        title: 'Always use named exports',
                        rule: 'All exports must be named, not default',
                        status: 'active',
                    },
                });

                const tools = buildConversationMemoryTools(
                    mockKodyRulesService as any,
                    ORG_ID,
                    TEAM_ID,
                );

                const result = await tools.KODUS_CREATE_MEMORY.execute({
                    title: 'Always use named exports',
                    rule: 'All exports must be named, not default',
                });

                const parsed = JSON.parse(result);
                // Observable behavior: same confirmation as legacy MCP path
                expect(parsed.success).toBe(true);
                expect(parsed.action).toBe('created');
                expect(parsed.requiresApproval).toBe(false);
                expect(parsed.link).toContain('memories');
            });

            it('implicit capture — model calls tool without explicit @kody remember', async () => {
                // The model decides to call KODUS_CREATE_MEMORY based on user intent
                // (e.g., "we always use AWS for cloud infrastructure")
                mockKodyRulesService.createOrUpdateMemory.mockResolvedValue({
                    action: 'created',
                    requiresApproval: false,
                    link: 'https://app.kodus.io/rules/org-regression-test/memories',
                    rule: { uuid: 'rule-uuid-456', status: 'active' },
                });

                const tools = buildConversationMemoryTools(
                    mockKodyRulesService as any,
                    ORG_ID,
                    TEAM_ID,
                );

                // Implicit: the tool's execute is called by the agent with inferred args
                const result = await tools.KODUS_CREATE_MEMORY.execute({
                    title: 'AWS is the cloud provider',
                    rule: 'All cloud infrastructure code should be compatible with AWS',
                });

                // Assert: createOrUpdateMemory was called with the correct organizationId
                // (prevents Pitfall 3 — wrong org)
                expect(
                    mockKodyRulesService.createOrUpdateMemory,
                ).toHaveBeenCalledWith(
                    expect.objectContaining({ organizationId: ORG_ID }),
                    expect.objectContaining({
                        title: 'AWS is the cloud provider',
                    }),
                    expect.any(Object),
                );

                const parsed = JSON.parse(result);
                expect(parsed.action).toBe('created');
            });

            it('duplicate detection — createOrUpdateMemory returns action:skipped → JSON has action:skipped', async () => {
                mockKodyRulesService.createOrUpdateMemory.mockResolvedValue({
                    action: 'skipped',
                    requiresApproval: false,
                    link: 'https://app.kodus.io/rules/org-regression-test/memories',
                    rule: {
                        uuid: 'rule-existing-789',
                        title: 'Always use named exports',
                        rule: 'All exports must be named, not default',
                        status: 'active',
                    },
                });

                const tools = buildConversationMemoryTools(
                    mockKodyRulesService as any,
                    ORG_ID,
                    TEAM_ID,
                );

                const result = await tools.KODUS_CREATE_MEMORY.execute({
                    title: 'Always use named exports',
                    rule: 'All exports must be named, not default',
                });

                const parsed = JSON.parse(result);
                // Duplicate detection: no new memory created, action is 'skipped'
                expect(parsed.action).toBe('skipped');
                // createOrUpdateMemory was still called (deduplication happens inside the service)
                expect(
                    mockKodyRulesService.createOrUpdateMemory,
                ).toHaveBeenCalledTimes(1);
            });

            it('pending approval — returns requiresApproval:true in observable output', async () => {
                mockKodyRulesService.createOrUpdateMemory.mockResolvedValue({
                    action: 'created',
                    requiresApproval: true,
                    link: 'https://app.kodus.io/rules/org-regression-test/memories',
                    rule: { uuid: 'rule-pending', status: 'pending' },
                });

                const tools = buildConversationMemoryTools(
                    mockKodyRulesService as any,
                    ORG_ID,
                    TEAM_ID,
                );

                const result = await tools.KODUS_CREATE_MEMORY.execute({
                    title: 'Pending rule',
                    rule: 'Requires centralized config approval',
                });

                const parsed = JSON.parse(result);
                expect(parsed.requiresApproval).toBe(true);
                expect(parsed.action).toBe('created');
            });
        });

        describe('KODUS_FIND_MEMORIES', () => {
            it('returns memories from service as JSON string', async () => {
                mockKodyRulesService.findMemories.mockResolvedValue({
                    success: true,
                    count: 1,
                    data: [
                        {
                            uuid: 'mem-1',
                            title: 'Always use named exports',
                            rule: 'All exports must be named',
                        },
                    ],
                });

                const tools = buildConversationMemoryTools(
                    mockKodyRulesService as any,
                    ORG_ID,
                    TEAM_ID,
                );

                const result = await tools.KODUS_FIND_MEMORIES.execute({});
                const parsed = JSON.parse(result);
                expect(parsed.success).toBe(true);
                expect(parsed.data).toHaveLength(1);
            });
        });
    });
    ```
