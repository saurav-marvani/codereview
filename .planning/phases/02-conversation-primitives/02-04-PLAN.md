<objective>
Build `buildConversationMemoryTools` — a factory function that wraps `IKodyRulesService.createOrUpdateMemory` and `IKodyRulesService.findMemories` as `mkTool`-compatible objects (Option A: Adapter). Returns `Record<string, any>` for spreading into `AgentLoopInput.additionalTools`.

Note on dependency: The adapter injects `IKodyRulesService` directly (via `KODY_RULES_SERVICE_TOKEN` from `KodyRulesModule`) rather than via `KodyRulesTools` from `McpModule`. `KodyRulesTools` is only provided when `API_MCP_SERVER_ENABLED=true`; `IKodyRulesService` is always available via `KodyRulesModule`. The factory is a plain function (not a NestJS service) — callers inject `IKodyRulesService` and pass it in.

    - Mirrors `mkTool` shape from `agent-tools.factory.ts`: `{ type: 'function', description, inputSchema: jsonSchema(schema), execute }`
    - Uses `jsonSchema()` from `ai` package (already in tree)

    ```typescript
    import { jsonSchema } from 'ai';
    import type { IKodyRulesService } from '@libs/kodyRules/domain/contracts/kodyRules.service.contract';
    import { KodyRulesType, KodyRulesOrigin, KodyRulesStatus } from '@libs/kodyRules/domain/enums/kodyRules.enum';

    /**
     * Build AI SDK-compatible tool objects for conversation memory operations.
     * Wraps IKodyRulesService.createOrUpdateMemory and findMemories as mkTool-compatible
     * objects for spreading into AgentLoopInput.additionalTools (EXT-03 / CONV-03).
     *
     * organizationId and teamId are passed per-call — never cached at build time.
     * This prevents Pitfall 3: memories being created in the wrong org.
     *
     * Returns {} when kodyRulesService is absent (safe fallback for self-hosted
     * instances where KodyRulesModule may not be wired).
     */
    export function buildConversationMemoryTools(
        kodyRulesService: IKodyRulesService | null | undefined,
        organizationId: string,
        teamId: string,
    ): Record<string, any> {
        if (!kodyRulesService) {
            return {};
        }

        const organizationAndTeamData = { organizationId, teamId };

        return {
            KODUS_CREATE_MEMORY: {
                type: 'function' as const,
                description:
                    'Capture a memory, preference, or coding rule derived from context. Invoke when the user demonstrates explicit or implicit intent to save a convention or preference. After execution, inform the user: (1) action taken (created/updated/skipped), (2) whether approval is required, (3) provided link.',
                inputSchema: jsonSchema({
                    type: 'object' as const,
                    properties: {
                        title: {
                            type: 'string',
                            description:
                                'Descriptive title for the memory rule',
                        },
                        rule: {
                            type: 'string',
                            description:
                                'Detailed description of the coding rule or convention to enforce',
                        },
                        repositoryId: {
                            type: 'string',
                            description:
                                'Optional: limit to a specific repository (omit for global)',
                        },
                        directoryId: {
                            type: 'string',
                            description:
                                'Optional: limit to a specific directory (requires repositoryId)',
                        },
                        path: {
                            type: 'string',
                            description:
                                'Optional: glob pattern to limit to specific files',
                        },
                    },
                    required: ['title', 'rule'],
                }),
                execute: async (args: {
                    title: string;
                    rule: string;
                    repositoryId?: string;
                    directoryId?: string;
                    path?: string;
                }): Promise<string> => {
                    const result =
                        await kodyRulesService.createOrUpdateMemory(
                            organizationAndTeamData,
                            {
                                title: args.title,
                                type: KodyRulesType.MEMORY,
                                rule: args.rule,
                                origin: KodyRulesOrigin.GENERATED,
                                status: KodyRulesStatus.ACTIVE,
                                repositoryId: args.repositoryId ?? 'global',
                                directoryId: args.directoryId ?? null,
                                path: args.path ?? null,
                            },
                            {
                                userId: 'kody-conversation-agent',
                                userEmail: 'kody@kodus.io',
                            },
                        );

                    const action = result?.action ?? 'created';
                    const requiresApproval = result?.requiresApproval ?? false;
                    const link = result?.link ?? '';
                    return JSON.stringify({
                        success: true,
                        action,
                        requiresApproval,
                        link,
                        message: `Memory ${action}. Approval required: ${requiresApproval ? 'yes' : 'no'}. Link: ${link}`,
                    });
                },
            },

            KODUS_FIND_MEMORIES: {
                type: 'function' as const,
                description:
                    'Find existing memories, preferences, and coding rules for this organization. Call at the start of each conversation to load relevant context.',
                inputSchema: jsonSchema({
                    type: 'object' as const,
                    properties: {
                        repositoryId: {
                            type: 'string',
                            description:
                                'Optional: filter memories by repository',
                        },
                        keywords: {
                            type: 'array',
                            items: { type: 'string' },
                            description: 'Optional: filter by keywords',
                        },
                        limit: {
                            type: 'number',
                            description: 'Optional: max results (default 20)',
                        },
                    },
                    required: [],
                }),
                execute: async (args: {
                    repositoryId?: string;
                    keywords?: string[];
                    limit?: number;
                }): Promise<string> => {
                    const result = await kodyRulesService.findMemories(
                        organizationAndTeamData,
                        {
                            repositoryId: args.repositoryId,
                            keywords: args.keywords,
                            limit: args.limit ?? 20,
                        },
                    );
                    return JSON.stringify(result);
                },
            },
        };
    }
    ```

    ```typescript
    import { buildConversationMemoryTools } from './conversation-tools.factory';

    describe('buildConversationMemoryTools', () => {
        const mockKodyRulesService = {
            createOrUpdateMemory: jest.fn(),
            findMemories: jest.fn(),
        };

        const ORG_ID = 'org-test-123';
        const TEAM_ID = 'team-test-456';

        beforeEach(() => {
            jest.clearAllMocks();
        });

        it('returns empty object when kodyRulesService is null', () => {
            const tools = buildConversationMemoryTools(null, ORG_ID, TEAM_ID);
            expect(tools).toEqual({});
        });

        it('returns KODUS_CREATE_MEMORY and KODUS_FIND_MEMORIES keys', () => {
            const tools = buildConversationMemoryTools(
                mockKodyRulesService as any,
                ORG_ID,
                TEAM_ID,
            );
            expect(tools).toHaveProperty('KODUS_CREATE_MEMORY');
            expect(tools).toHaveProperty('KODUS_FIND_MEMORIES');
        });

        it('KODUS_CREATE_MEMORY.execute calls createOrUpdateMemory with correct orgId', async () => {
            mockKodyRulesService.createOrUpdateMemory.mockResolvedValue({
                action: 'created',
                requiresApproval: false,
                link: 'https://app.kodus.io/memories/123',
                rule: {},
            });

            const tools = buildConversationMemoryTools(
                mockKodyRulesService as any,
                ORG_ID,
                TEAM_ID,
            );

            await tools.KODUS_CREATE_MEMORY.execute({
                title: 'Always use named exports',
                rule: 'All exports must be named, not default',
            });

            expect(
                mockKodyRulesService.createOrUpdateMemory,
            ).toHaveBeenCalledWith(
                expect.objectContaining({ organizationId: ORG_ID }),
                expect.objectContaining({
                    title: 'Always use named exports',
                }),
                expect.any(Object),
            );
        });

        it('KODUS_CREATE_MEMORY.execute returns JSON string with action and link', async () => {
            mockKodyRulesService.createOrUpdateMemory.mockResolvedValue({
                action: 'skipped',
                requiresApproval: false,
                link: 'https://app.kodus.io/memories/456',
                rule: {},
            });

            const tools = buildConversationMemoryTools(
                mockKodyRulesService as any,
                ORG_ID,
                TEAM_ID,
            );

            const result = await tools.KODUS_CREATE_MEMORY.execute({
                title: 'Duplicate rule',
                rule: 'already exists',
            });

            const parsed = JSON.parse(result);
            expect(parsed.action).toBe('skipped');
        });

        it('KODUS_FIND_MEMORIES.execute calls findMemories with correct orgId', async () => {
            mockKodyRulesService.findMemories.mockResolvedValue({
                success: true,
                data: [],
            });

            const tools = buildConversationMemoryTools(
                mockKodyRulesService as any,
                ORG_ID,
                TEAM_ID,
            );

            await tools.KODUS_FIND_MEMORIES.execute({});

            expect(mockKodyRulesService.findMemories).toHaveBeenCalledWith(
                expect.objectContaining({ organizationId: ORG_ID }),
                expect.any(Object),
            );
        });
    });
    ```
