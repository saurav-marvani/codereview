    Create `test/unit/platform/use-cases/byok-queue-timeout.spec.ts`.

    This test characterizes the BYOK queue timeout behavior: when a BYOK slot is held (review), a conversation call with `queueTimeoutMs: 60_000` fails with `[BYOK-QUEUE-TIMEOUT]` error after the timeout.

    The test should NOT import the full NestJS module. Test `runWithBYOKLimiter` directly (it's exported from `byok-to-vercel.ts`) with a mock BYOK config that sets `maxConcurrentRequests: 1`.

    ```typescript
    /**
     * PERF-03 Characterization Test: BYOK concurrency queue timeout for conversation path.
     *
     * This test documents the deadlock risk (review holds slot forever under infinite wait)
     * and confirms the mitigation (queueTimeoutMs: 60_000 causes conversation to fail fast
     * with [BYOK-QUEUE-TIMEOUT] rather than hanging indefinitely).
     */
    import { runWithBYOKLimiter } from '@libs/code-review/infrastructure/agents/llm/byok-to-vercel';

    const MOCK_BYOK_CONFIG = {
        main: {
            provider: 'openai' as const,
            apiKey: 'test-key',
            baseURL: 'https://api.openai.com/v1',
            model: 'gpt-4',
            maxConcurrentRequests: 1,
        },
    };

    describe('PERF-03: BYOK queue timeout characterization', () => {
        it('conversation call with queueTimeoutMs times out when review holds the slot', async () => {
            // Simulate review holding the slot (never resolves during test window)
            let reviewResolve: () => void;
            const reviewHoldingSlot = new Promise<void>((resolve) => {
                reviewResolve = resolve;
            });

            // Review call: acquires the single BYOK slot, holds it
            const reviewCallStarted = new Promise<void>((resolve) => {
                void runWithBYOKLimiter(
                    {
                        byokConfig: MOCK_BYOK_CONFIG,
                        queueTimeoutMs: 0, // review: no timeout (infinite wait is OK for review)
                    },
                    async () => {
                        resolve(); // signal that review has the slot
                        await reviewHoldingSlot;
                        return 'review-done';
                    },
                    'review-call',
                ).catch(() => {}); // silence any errors on cleanup
            });

            // Wait for review to hold the slot
            await reviewCallStarted;

            // Conversation call: tries to acquire the occupied slot with 100ms timeout (short for test speed)
            const conversationError = await runWithBYOKLimiter(
                {
                    byokConfig: MOCK_BYOK_CONFIG,
                    queueTimeoutMs: 100, // short for test; production uses 60_000
                },
                async () => 'should-not-run',
                'conversation-call',
            ).catch((err: Error) => err);

            // Confirm: conversation timed out with the expected error
            expect(conversationError).toBeInstanceOf(Error);
            expect((conversationError as Error).message).toContain('[BYOK-QUEUE-TIMEOUT]');

            // Cleanup: release the review slot
            reviewResolve!();
        }, 5_000); // 5s Jest timeout (100ms queue timeout + slack)

        it('conversation succeeds when BYOK slot is free', async () => {
            const result = await runWithBYOKLimiter(
                {
                    byokConfig: MOCK_BYOK_CONFIG,
                    queueTimeoutMs: 60_000,
                },
                async () => 'conversation-reply',
                'conversation-call-free-slot',
            );

            expect(result).toBe('conversation-reply');
        });
    });
    ```

    Note: The test uses `queueTimeoutMs: 100` (not 60_000) for test speed — it characterizes the MECHANISM, not the specific duration. A comment explains the production value.
