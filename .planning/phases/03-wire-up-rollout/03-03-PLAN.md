    private async handleConversation(context: {
        prepareContext: any;
        organizationAndTeamData: OrganizationAndTeamData;
        thread: any;
    }): Promise<string> {
        const { prepareContext, organizationAndTeamData, thread } = context;

        // --- Feature-flag dispatch (RLLT-01, RLLT-02) ---
        // Flag is evaluated ONCE per webhook invocation. In-flight calls complete
        // on the path they started; a flag flip affects the NEXT webhook only.
        const featureIdentifier =
            organizationAndTeamData.organizationId ?? 'unknown';
        const repositoryId: string | undefined = prepareContext.repository?.id;

        const envOverride =
            process.env.API_CONVERSATION_RUNTIME_ENABLED?.toLowerCase();
        let useNewRuntime =
            envOverride === 'true' || envOverride === '1';

        // Self-hosted note: when posthog.isInitialized === false (no PostHog key
        // configured) AND the env override is absent, useNewRuntime stays false
        // and the legacy path runs. This is the intended self-hosted default.
        if (!useNewRuntime && posthog.isInitialized) {
            const flagResult = await posthog.isFeatureEnabled(
                FEATURE_FLAGS.conversationAgentRuntime,
                featureIdentifier,
                organizationAndTeamData,
                repositoryId,
            );
            useNewRuntime = flagResult === true;
        }

        if (useNewRuntime) {
            return await this.handleConversationViaRuntime(context);
        }

        // Legacy fallback — unchanged (RLLT-02)
        return await this.conversationAgentUseCase.execute({
            prompt: prepareContext.userQuestion,
            organizationAndTeamData,
            prepareContext: prepareContext,
            thread: thread,
        });
    }
    ```

    private async handleConversationViaRuntime(context: {
        prepareContext: any;
        organizationAndTeamData: OrganizationAndTeamData;
        thread: any;
    }): Promise<string> {
        const { prepareContext, organizationAndTeamData } = context;

        const prKey = buildPrKey(
            organizationAndTeamData.organizationId,
            prepareContext.repository?.id ?? 'unknown',
            prepareContext.pullRequest?.pullRequestNumber ?? 0,
        );

        // Acquire sandbox lease (5 min TTL covers LLM call + comment posting)
        const { sandbox, leaseId, wasCreated } =
            await this.leaseManager.acquire(prKey, 'conversation', 5 * 60 * 1000);

        // Resolve BYOK config and build a Vercel AI SDK LanguageModel.
        // byokToVercelModel() accepts undefined and falls back to the Kodus
        // default model (Gemini for cloud, API_LLM_PROVIDER_MODEL for self-hosted).
        // This mirrors the pattern in base-code-review-agent.provider.ts lines 475-480.
        const byokConfig = await this.permissionValidationService.getBYOKConfig(
            organizationAndTeamData,
        );
        const model = byokToVercelModel(byokConfig ?? undefined);
        // byokToVercelModel is imported from:
        // @libs/code-review/infrastructure/agents/llm/byok-to-vercel

        // Phase 4 instrumentation label (measurement deferred to Phase 4)
        const sandboxState =
            sandbox.type === 'null'
                ? 'null'
                : wasCreated
                  ? 'cold-create'
                  : 'paused-resumed';

        try {
            const result = await runConversationLoop({
                model,
                systemPrompt: CONVERSATION_SYSTEM_PROMPT,
                userPrompt: prepareContext.userQuestion ?? '',
                prKey,
                sandbox,
                sessionManager: this.sessionManager,
                memoryTools: buildConversationMemoryTools(
                    this.kodyRulesService,
                    organizationAndTeamData.organizationId,
                    organizationAndTeamData.teamId,
                ),
                byokConfig: byokConfig ?? undefined,
                repositoryFullName: prepareContext.repository?.fullName,
            });

            return result.reply;
        } catch (err) {
            // Known BYOK errors get a specific user message; all others get generic
            const message =
                err instanceof Error &&
                (err.message.includes('[BYOK-QUEUE-TIMEOUT]') ||
                    err.message.includes('[BYOK-QUEUE-ABORTED]'))
                    ? 'Kody is processing another request for this repository — please try again in a moment.'
                    : 'Kody encountered an error processing your request. Please try again.';

            // Structured log for debugging (sandboxState for Phase 4)
            const logger = (this as any).logger ?? console;
            if (typeof logger.error === 'function') {
                logger.error({
                    message: 'handleConversationViaRuntime error',
                    context: 'ChatWithKodyFromGitUseCase',
                    metadata: {
                        prKey,
                        sandboxState,
                        byokProvider: byokConfig?.main?.provider ?? 'kodus-default',
                        error: err instanceof Error ? err.message : String(err),
                    },
                });
            }

            // Return user-visible string — handleConversationFlow posts it (Pitfall 7: do NOT throw)
            return message;
        } finally {
            // MANDATORY: release even if runConversationLoop throws (Pitfall 1: lease leak prevention)
            await this.leaseManager.release(leaseId);
        }
    }
    ```
